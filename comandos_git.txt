-(Inicializar la carpeta como proyectoGIT)

git config --global user.name "adeir"

git add .

-(descartar cambios de algun archivo sin hacer commit o add)
git checkout  -- uno.txt

git log
-(diferenciar cambios)
git diff
-(listar branchs)
git branch

git branch adeir
-(cambio de rama o branch)
git checkout adeir
